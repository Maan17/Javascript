    Example of Conditional Code

  • if(10 < 20) { alert('Hellow student'); }

    Else If

  • if(10 < 20) { alert('Hellow student'); }else if(200 < 400) { alert('Of course 200 is smaller'); }else { alert('everything is just false'); }

    While Loop

    var counter = 0;
  • while(10 < 20) { alert('Hellow student I hope you like this'); counter++; }

    Do While Loop

    var counter = 0;
  • do{ alert('If you made it this far, you have learned!'); counter++; }(10 < 20);

    For Loop

  • for(var i =0; i < 10; i++){ alert('Yesssss'); }

    Function with No Parameters

  • function name() { alert("hello from the functions name"); } name();

    Function Passing values

    function name(variable1,variable2) { var calculation = variable1 + variable2 document.write(calculation); } name(10,14);

    Function Returning values

    function name() { var variable1 = 100; var variable2 = 200; var calculation = variable1 + variable2 return calculation; } var result = name(10,14); alert(result);
